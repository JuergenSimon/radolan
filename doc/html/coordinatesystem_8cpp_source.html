00127         {
00128                 // distance from origin
00129                 double dx,dy;
00130 
00131                 // find out what quadrant we're in:
00132                 RDGridQuadrant quadrant = this->RDQuadrant(p);
00133                 switch (quadrant) 
00134                 {
00135                         case RDLowerLeft:
00136                                 dx = p.ix - (m_offset.x - 1.0f) - 1.0f;
00137                                 dy = p.iy - (m_offset.y - 1.0f) - 1.0f;
00138                                 break;
00139                         case RDUpperLeft:
00140                                 dx = p.ix - (m_offset.x - 1.0f) - 1.0f;
00141                                 dy = p.iy - (m_offset.y - 1.0f);
00142                                 break;
00143                         case RDUpperRight:
00144                                 dx = p.ix - (m_offset.x - 1.0f);
00145                                 dy = p.iy - (m_offset.y - 1.0f);
00146                                 break;
00147                         case RDLowerRight:
00148                                 dx = p.ix - (m_offset.x - 1.0f);
00149                                 dy = p.iy - (m_offset.y - 1.0f) - 1.0f;
00150                                 break;
00151                 }
00152                 
00153                 // to get the real distance, multiply with mesh width
00154                 dx *= MESH_WIDTH;

00160         RDGeographicalPointRad RDCoordinateSystem::geographicalCoordinateRad(RDCartesianPoint p)
00161         {
00162                 static double b_lambda_0 = rad(LAMBDA_0);       
00163                 static double b_phi_0 = rad(PHI_0);     
00164                 RDGeographicalPointRad pg;
00165                 pg.longitude = b_lambda_0 + atan(-p.x/p.y);
00166                 double omphi = R_EARTH * R_EARTH * (1+sin(b_phi_0)) * (1+sin(b_phi_0));
00167                 double xy = (p.x*p.x+p.y*p.y);
00168                 pg.latitude = asin((omphi-xy)/(omphi+xy));
00169                 return pg;
00170         }

00266                 else if (dx>=0.0 && dy>=0.0)

00268                 else if (dx>=0.0 && dy<0.0)

00271                 // count

00277                                                                 (int)(floor(dy) + m_offset.y));
00278                                 break;

00280                                 p = rdGridPoint((int)(floor(dx) + m_offset.x),
00281                                                                 (int)(ceil(dy) + m_offset.y - 1.0f));
00282                                 break;

00285                                                                 (int)(ceil(dy) + m_offset.y - 1.0f));

00289                                                                 (int)(floor(dy) + m_offset.y));
