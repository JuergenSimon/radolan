103  RDCartesianPoint RDCoordinateSystem::cartesianCoordinate(RDGeographicalPointRad p)
104  {
105  // Bezugskoordinate im Bogenmass
106  static double lambda0 = rad(LAMBDA_0);
107  static double phi0 = rad(PHI_0);
108  double M = polarStereographicScalingFactor(phi0,p.latitude);
109  return rdCartesianPoint( R_EARTH * M * cos(p.latitude) * sin(p.longitude-lambda0),
110  - R_EARTH * M * cos(p.latitude) * cos(p.longitude-lambda0));
111  }

120  RDGridQuadrant RDCoordinateSystem::RDQuadrant(RDGridPoint p)
121  {
122  bool isLeft = (p.ix - m_offset.x <= 0);
123 
124  bool isLower = (p.iy - m_offset.y <= 0);
125 
126  RDGridQuadrant result;
127 
128  if (isLeft && isLower)
129  {
130  result = RDLowerLeft;
131  }
132  else if (isLeft && !isLower)
133  {
134  result = RDUpperLeft;
135  }
136  else if (!isLeft && !isLower)
137  {
138  result = RDUpperRight;
139  }
140  else
141  {
142  result = RDLowerRight;
143  }
144  return result;
145  }

147  RDCartesianPoint RDCoordinateSystem::cartesianCoordinate(RDGridPoint p)
148  {
149  // distance from origin
150  double dx,dy;
151 
152  // find out what quadrant we're in:
153  RDGridQuadrant quadrant = this->RDQuadrant(p);
154  switch (quadrant)
155  {
156  case RDLowerLeft:
157  dx = p.ix - (m_offset.x - 1.0f) - 1.0f;
158  dy = p.iy - (m_offset.y - 1.0f) - 1.0f;
159  break;
160  case RDUpperLeft:
161  dx = p.ix - (m_offset.x - 1.0f) - 1.0f;
162  dy = p.iy - (m_offset.y - 1.0f);
163  break;
164  case RDUpperRight:
165  dx = p.ix - (m_offset.x - 1.0f);
166  dy = p.iy - (m_offset.y - 1.0f);
167  break;
168  case RDLowerRight:
169  dx = p.ix - (m_offset.x - 1.0f);
170  dy = p.iy - (m_offset.y - 1.0f) - 1.0f;
171  break;
172  }
173 
174  // to get the real distance, multiply with mesh width
175  dx *= MESH_WIDTH;
176  dy *= MESH_WIDTH;
177 
178  return rdCartesianPoint(m_originCartesian.x + dx,m_originCartesian.y + dy);
179  }

181  RDGeographicalPointRad RDCoordinateSystem::geographicalCoordinateRad(RDCartesianPoint p)
182  {
183  static double b_lambda_0 = rad(LAMBDA_0);
184  static double b_phi_0 = rad(PHI_0);
185  RDGeographicalPointRad pg;
186  pg.longitude = b_lambda_0 + atan(-p.x/p.y);
187  double omphi = R_EARTH * R_EARTH * (1+sin(b_phi_0)) * (1+sin(b_phi_0));
188  double xy = (p.x*p.x+p.y*p.y);
189  pg.latitude = asin((omphi-xy)/(omphi+xy));
190  return pg;
191  }

277  RDGridPoint RDCoordinateSystem::gridPoint(RDCartesianPoint g, bool& isInside)
278  {
279  // calculate the distance of the point from the origin
280  double dx = g.x - m_originCartesian.x;
281  double dy = g.y - m_originCartesian.y;
282 
283  // quadrant?
284  RDGridQuadrant quadrant = RDUpperRight;
285 
286  if (dx<0.0 && dy<0.0)
287  {
288  quadrant = RDLowerLeft;
289  }
290  else if (dx<0.0 && dy>=0.0)
291  {
292  quadrant = RDUpperLeft;
293  }
294  else if (dx>=0.0 && dy>=0.0)
295  {
296  quadrant = RDUpperRight;
297  }
298  else if (dx>=0.0 && dy<0.0)
299  {
300  quadrant = RDLowerRight;
301  }
302 
303  // count
304  RDGridPoint p;
305  switch (quadrant)
306  {
307  case RDLowerLeft:
308  p = rdGridPoint((int)(floor(dx) + m_offset.x),
309  (int)(floor(dy) + m_offset.y));
310  break;
311  case RDUpperLeft:
312  p = rdGridPoint((int)(floor(dx) + m_offset.x),
313  (int)(ceil(dy) + m_offset.y - 1.0f));
314  break;
315  case RDUpperRight:
316  p = rdGridPoint((int)(ceil(dx) + m_offset.x - 1.0f),
317  (int)(ceil(dy) + m_offset.y - 1.0f));
318  break;
319  case RDLowerRight:
320  p = rdGridPoint((int)(floor(dx) + m_offset.x - 1.0f),
321  (int)(floor(dy) + m_offset.y));
322  }
323 
324  if (p.ix>=0 && p.iy>=0 && p.ix < m_radolanGridCountHorizontal && p.iy < m_radolanGridCountVertical)
325  {
326  isInside = true;
327  } else {
328  isInside = false;
329  }
330 
331  return p;
332  }
